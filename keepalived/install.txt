apt install keepalived

#node 1
vi /etc/keepalived/keepalived.conf
vrrp_script chk_remote {
  script "/etc/keepalived/check_remote_services.sh"
  interval 5
  weight -20
}

vrrp_instance VI_1 {
  interface ens192
  state MASTER
  virtual_router_id 51
  priority 120
  virtual_ipaddress {
    10.0.139.200
  }
  track_script {
    chk_remote
  }
}
